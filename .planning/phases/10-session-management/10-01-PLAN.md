       ```python
       from __future__ import annotations
       from typing import Any
       from app.infra.database import db
       from app.infra import task_bus

       REJECT_PREFIX = "用户已主动取消"   # 与 frontend/src/app/page.tsx:65 的 reject 模板关键前缀对齐

       def messages_to_segments(raw_messages: list[Any]) -> list[dict]:
           """把 LangGraph checkpoint 里的 BaseMessage 列表映射到前端 Message[]。
           
           - 用 duck-typing (type(msg).__name__) 避开 conftest.py 对 langchain_core 的 MagicMock。
           - 历史 **不还原 HITL segment**（P-01）：tool_call 只还原为 tool pill，status=done/rejected。
           """
           result: list[dict] = []
           current_ai: dict | None = None   # 当前正在拼装的 assistant message
           for msg in raw_messages:
               tname = type(msg).__name__
               content = getattr(msg, "content", "")
               if isinstance(content, list):
                   content = "".join(
                       c if isinstance(c, str) else c.get("text", "") if isinstance(c, dict) else ""
                       for c in content
                   )
               if tname == "HumanMessage":
                   current_ai = None
                   result.append({
                       "id": f"user-{getattr(msg, 'id', len(result))}",
                       "role": "user",
                       "segments": [{"type": "text", "content": content or ""}],
                       "timestamp": 0,
                   })
               elif tname == "AIMessage":
                   if current_ai is None:
                       current_ai = {
                           "id": f"assistant-{getattr(msg, 'id', len(result))}",
                           "role": "assistant",
                           "segments": [],
                           "timestamp": 0,
                       }
                       result.append(current_ai)
                   if content:
                       current_ai["segments"].append({"type": "text", "content": content})
                   for tc in (getattr(msg, "tool_calls", None) or []):
                       tool_name = tc.get("name") if isinstance(tc, dict) else getattr(tc, "name", None)
                       if tool_name:
                           current_ai["segments"].append({"type": "tool", "name": tool_name, "status": "done"})
               elif tname == "ToolMessage":
                   if current_ai is None:
                       # 保守处理：未知场景，直接起一个新 assistant bubble 避免丢 tool
                       current_ai = {"id": f"assistant-{len(result)}", "role": "assistant", "segments": [], "timestamp": 0}
                       result.append(current_ai)
                   # 匹配到对应 tool name 的最后一个 tool segment，若内容含 reject 前缀则改 rejected
                   tool_name = getattr(msg, "name", None) or ""
                   is_reject = isinstance(content, str) and content.startswith(REJECT_PREFIX)
                   if is_reject:
                       for seg in reversed(current_ai["segments"]):
                           if seg.get("type") == "tool" and seg.get("name") == tool_name:
                               seg["status"] = "rejected"
                               break
           return result

       async def load_history_for_session(user_id: str, session_id: str, session_svc) -> dict:
           """读 checkpoint + session.last_task_id 组装 {messages, active_task, truncate_after_active_task}."""
           # active_task 来源：session.last_task_id → task_bus.get_task_meta
           session = await session_svc.get_session(session_id, user_id)
           active_task: dict | None = None
           if session:
               last_tid = session.get("last_task_id")
               if last_tid:
                   meta = await task_bus.get_task_meta(last_tid)
                   if meta and meta.get("status") in (task_bus.STATUS_RUNNING, task_bus.STATUS_INTERRUPTED):
                       active_task = {"task_id": last_tid, "status": meta["status"]}

           if db.checkpointer is None:
               return {"messages": [], "active_task": active_task, "truncate_after_active_task": False}
           config = {"configurable": {"thread_id": session_id}}
           ckpt = await db.checkpointer.aget_tuple(config)
           if ckpt is None:
               return {"messages": [], "active_task": active_task, "truncate_after_active_task": False}
           raw = ckpt.checkpoint.get("channel_values", {}).get("messages", []) or []
           truncate = bool(
               active_task is not None
               and raw
               and type(raw[-1]).__name__ == "AIMessage"
           )
           return {
               "messages": messages_to_segments(raw),
               "active_task": active_task,
               "truncate_after_active_task": truncate,
           }
       ```
