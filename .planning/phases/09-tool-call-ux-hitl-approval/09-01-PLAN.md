```typescript
export async function resumeChat(
  taskId: string,
  responseType: "approve" | "reject",
  message?: string
): Promise<InvokeResponse> {
  const body: Record<string, unknown> = {
    task_id: taskId,
    response_type: responseType,
  };
  if (responseType === "reject" && message) {
    body.args = { message };
  }
  const response = await fetch(`${API_BASE}/api/chat/resume`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body),
  });
  if (!response.ok) {
    const text = await response.text();
    throw new Error(text || `HTTP ${response.status}`);
  }
  return response.json() as Promise<InvokeResponse>;
}
```

```typescript
eventSource.addEventListener("hitl", (event) => {
  const payload = JSON.parse((event as MessageEvent).data);
  // 从 interrupt_value 提取工具名和描述
  // 后端 hitl 事件 data 结构来自 LangGraph __interrupt__
  // 典型: { action_requests: [{ name: "tool_name", args: {...} }] }
  const actionReq = payload.action_requests?.[0];
  const toolName = actionReq?.name ?? "unknown";
  // 生成自然语言描述（per D-05: 不展示原始 JSON）
  const description = formatHitlDescription(toolName, actionReq?.args);
  addHitlSegment(toolName, description, taskId);
  setStatus("interrupted");
});
```

3. 在 `use-sse.ts` 文件顶层（hook 函数之外）添加一个纯函数 `formatHitlDescription`：

```typescript
function formatHitlDescription(toolName: string, args?: Record<string, unknown>): string {
  if (!args || Object.keys(args).length === 0) {
    return `Agent 想要调用 ${toolName}`;
  }
  // 提取前两个参数的值作为摘要
  const entries = Object.entries(args).slice(0, 2);
  const summary = entries
    .map(([, v]) => (typeof v === "string" ? v : JSON.stringify(v)))
    .join("、");
  return `Agent 想要调用 ${toolName}：${summary}`;
}
```

4. 当收到 hitl 事件时，EventSource 不需要关闭 — 后端 hitl 事件之后不会再发新事件（task 状态变为 interrupted），但 SSE stream 的 read_events 会阻塞等待。resume 后后端继续往同一 stream 写入事件，EventSource 仍在连接中可以继续接收。这是正确的行为（per D-09: 同一条 SSE 连接续接）。

5. 将 `addHitlSegment` 加入 useEffect 的依赖数组。

6. 注意 hitl 事件不调用 `finishMessage` — 消息未结束，只是暂停等待审批。status 设为 "interrupted" 而非 "idle"。
