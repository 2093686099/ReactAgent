```tsx
export default function ChatPage() {
  // Store selectors — 分开订阅避免不必要的重渲染
  const messages = useChatStore(s => s.messages);
  const status = useChatStore(s => s.status);
  const currentTaskId = useChatStore(s => s.currentTaskId);
  const activeSessionId = useChatStore(s => s.activeSessionId);

  // Store actions
  const addUserMessage = useChatStore(s => s.addUserMessage);
  const addAssistantMessage = useChatStore(s => s.addAssistantMessage);
  const setStatus = useChatStore(s => s.setStatus);
  const setCurrentTaskId = useChatStore(s => s.setCurrentTaskId);
  const setError = useChatStore(s => s.setError);

  // Hooks
  useSSE(currentTaskId);   // SSE 连接管理
  const scrollRef = useRef<HTMLDivElement>(null);
  const { scrollToBottom, onScroll, shouldAutoScroll } = useAutoScroll(scrollRef);

  // 自动滚动：messages 或 status 变化时
  useEffect(() => {
    if (shouldAutoScroll) {
      requestAnimationFrame(() => scrollToBottom());
    }
  }, [messages, status, shouldAutoScroll, scrollToBottom]);

  // 发送消息处理
  const handleSend = async (text: string) => {
    addUserMessage(text);
    setStatus('sending');
    addAssistantMessage();  // 创建空的 assistant 消息占位

    try {
      const res = await invokeChat(activeSessionId, text);
      setCurrentTaskId(res.task_id);
      setStatus('streaming');
    } catch (err) {
      // 发送失败 — 移除空的 assistant 消息，恢复用户输入
      // 注意：store 需要提供 removeLastMessage action 或在 setError 中处理
      setError(err instanceof Error ? err.message : '未知错误');

      // 根据错误类型显示不同 toast（per UI-SPEC CHAT-07）
      if (err instanceof TypeError) {
        // 网络错误（TypeError: Failed to fetch）
        toast.error('发送失败，请检查网络连接');
      } else {
        toast.error('服务暂时不可用，请稍后重试');
      }
    }
  };

  return (
    <AppLayout>
      <ChatArea>
        <MessageList
          messages={messages}
          status={status}
          scrollRef={scrollRef}
          onScroll={onScroll}
        />
        <ChatInput onSend={handleSend} />
      </ChatArea>
    </AppLayout>
  );
}
```

3. **错误处理完善**（per CHAT-07, UI-SPEC Error State）：
   - 导入 sonner 的 `toast` 函数
   - POST /invoke 失败时：toast 提示（如上），用户消息不从 input 清除（ChatInput 内部在 onSend 回调成功后才清除，需要确认 ChatInput 实现是否已处理——如果 onSend 是同步调用，需要调整为 await 或先清除后失败时恢复）
   - 实现策略：ChatInput 的 onSend 触发后立即清空 input（乐观更新），如果 handleSend 失败，用户需要重新输入。这符合聊天应用的通用 UX 模式。
   - SSE error 事件的错误已在 useSSE hook 中处理（Phase 08 静默处理连接断开）。如果后端发送 SSE error 事件类型，useSSE 会调用 setError。在 page.tsx 中监听 status 变化为 'error' 时，不需要额外处理（错误信息已由 store 管理，MessageList 或 MessageBubble 会根据 status 展示 inline error）
