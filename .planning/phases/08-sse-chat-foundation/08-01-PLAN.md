export type TextSegment = { type: 'text'; content: string };
export type ToolSegment = { type: 'tool'; name: string; status: 'calling' | 'done' };

export interface Message {
  id: string;
  role: 'user' | 'assistant';
  segments: Segment[];

}

export type ChatStatus = 'idle' | 'sending' | 'streaming' | 'error';

2. **创建 `frontend/src/lib/api.ts`**：
```typescript
const API_BASE = 'http://localhost:8001';

export async function invokeChat(sessionId: string, query: string): Promise<InvokeResponse> {
  const res = await fetch(`${API_BASE}/api/chat/invoke`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ session_id: sessionId, query }),
  });
  if (!res.ok) {
    const text = await res.text();
    throw new Error(text || `HTTP ${res.status}`);
  }
  return res.json();
}
```
导出 `API_BASE` 常量供 useSSE 使用。不使用 axios（per STACK.md "What NOT to Use"）。

3. **创建 `frontend/src/stores/chat-store.ts`**（per D-14 domain-sliced, per D-13 RAF batch）：
- 使用 Zustand `create` 创建 store
- State 字段：`messages: Message[]`, `status: ChatStatus`, `currentTaskId: string | null`, `activeSessionId: string`（默认生成一个 UUID）
- 内部 RAF 批处理机制：用闭包维护 `tokenBuffer: string` 和 `rafId: number`
  - `appendToken(text)`: 将 text 追加到 buffer，如果 rafId 为 0 则启动 `requestAnimationFrame(flushTokenBuffer)`
  - `flushTokenBuffer()`: 将 buffer 内容合并到最后一条 assistant 消息的最后一个 text segment 的 content 中，清空 buffer，rafId = 0
- Actions：
  - `addUserMessage(text: string)`: 创建用户消息（`id: crypto.randomUUID()`, role: 'user', segments: [{type: 'text', content: text}], timestamp: Date.now()），追加到 messages
  - `addAssistantMessage()`: 创建空的 assistant 消息（segments: [{type: 'text', content: ''}]），追加到 messages
  - `appendToken(text: string)`: RAF 批处理追加 token 到最后一条 assistant 消息
  - `addToolSegment(name: string, status: string)`: 在最后一条 assistant 消息中 push 新的 tool segment；然后 push 新的空 text segment（为后续 token 准备）
  - `updateToolSegment(name: string, status: string)`: 找到最后一条 assistant 消息中匹配 name 的 tool segment，更新 status
  - `finishMessage()`: flush 残余 buffer，设 status 为 'idle'，清空 currentTaskId
  - `setError(message: string)`: 设 status 为 'error'
  - `setStatus(status: ChatStatus)`: 设状态
  - `setCurrentTaskId(taskId: string | null)`: 设当前 task ID
  - `reset()`: 清空 messages，status = 'idle'，currentTaskId = null

4. **创建 `frontend/src/hooks/use-sse.ts`**（per D-12 使用原生 EventSource）：
- 接收参数 `taskId: string | null`
- 从 store 获取 actions：`appendToken`, `addToolSegment`, `updateToolSegment`, `finishMessage`, `setError`, `setStatus`
- `useEffect` 依赖 `[taskId]`：
  - 如果 taskId 为 null，return
  - 创建 `const es = new EventSource(\`${API_BASE}/api/chat/stream/${taskId}?from_id=0\`)`
  - `es.addEventListener('token', (e) => { const { text } = JSON.parse(e.data); appendToken(text); })`
  - `es.addEventListener('tool', (e) => { const { name, status } = JSON.parse(e.data); if (status === 'calling') addToolSegment(name, status); else updateToolSegment(name, status); })`
  - `es.addEventListener('done', (e) => { finishMessage(); es.close(); })`
  - `es.addEventListener('error', (e) => { /* 如果 e 是 SSE error event（有 data）则解析 message 并 setError；否则是连接错误，静默关闭（Phase 12 做重连） */ })`
  - `es.onerror = () => { /* 连接错误，Phase 08 不做重连，直接 close + setStatus('idle') */ es.close(); }`
  - Cleanup：`return () => { es.close(); }`
- 注意：EventSource `addEventListener('error', ...)` 和 `es.onerror` 的区别 — 后端的 error 事件类型通过 addEventListener 捕获，连接断开通过 onerror 捕获

5. **创建 `frontend/src/hooks/use-auto-scroll.ts`**：
- 接收一个 `ref: RefObject<HTMLDivElement>` 参数（message list 的 scrollable container）
- 内部维护 `isUserScrolledUp: boolean` 状态
- 提供 `scrollToBottom()` 方法，用 `ref.current.scrollTop = ref.current.scrollHeight`
- 提供 `onScroll` handler：检查 `scrollHeight - scrollTop - clientHeight > 100`，如果是则设 `isUserScrolledUp = true`，否则 `isUserScrolledUp = false`
- 提供 `shouldAutoScroll` 布尔值（`!isUserScrolledUp`）
- 返回 `{ scrollToBottom, onScroll, shouldAutoScroll }`
