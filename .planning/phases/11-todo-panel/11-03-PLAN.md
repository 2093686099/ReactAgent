    按 RESEARCH.md §Pattern 3 + PATTERNS.md §ui-store.ts 骨架：

    ```typescript
    import { create } from "zustand";
    import { persist, createJSONStorage } from "zustand/middleware";

    type UIState = {
      todoDrawerOpen: boolean;
      hasAutoOpenedFor: Set<string>;
      toggleDrawer: () => void;
      openDrawer: () => void;
      closeDrawer: () => void;
      autoOpenDrawer: (sessionId: string) => void;
    };

    export const useUIStore = create<UIState>()(
      persist(
        (set) => ({
          todoDrawerOpen: false,
          hasAutoOpenedFor: new Set<string>(),
          toggleDrawer: () => set((s) => ({ todoDrawerOpen: !s.todoDrawerOpen })),
          openDrawer: () => set({ todoDrawerOpen: true }),
          closeDrawer: () => set({ todoDrawerOpen: false }),
          autoOpenDrawer: (sessionId) =>
            set((s) => {
              if (s.hasAutoOpenedFor.has(sessionId)) return s;
              const next = new Set(s.hasAutoOpenedFor);
              next.add(sessionId);
              return { hasAutoOpenedFor: next, todoDrawerOpen: true };
            }),
        }),
        {
          name: "neuron-assistant:ui-store:v1",
          storage: createJSONStorage(() => localStorage),
          skipHydration: true,                            // SSR mismatch 规避
          partialize: (s) => ({ todoDrawerOpen: s.todoDrawerOpen }) as Partial<UIState>,
        },
      ),
    );
    ```

    **关键点：**
    - `hasAutoOpenedFor: Set` 不入 persist（partialize 过滤）—— 防止"跨天重开仍记得已弹过"的反直觉（RESEARCH.md A2）
    - `autoOpenDrawer` 的幂等 Set.has 检查同时守住 user close 语义：用户手动 close 后，Set 里仍含该 sessionId，再次 auto-open 被拦截，drawer 保持 closed
    - 项目**首次**落地 Zustand persist —— 本模式应作为后续 UI 持久化范式（VALIDATION.md 已锁定 key 命名）

        autoOpenDrawer("s2");       // different sessionId → opens
        const s = useUIStore.getState();
        expect(s.todoDrawerOpen).toBe(true);
        expect(s.hasAutoOpenedFor.has("s2")).toBe(true);

| SSE server → Browser | EventSource `todo` 事件携带 payload `{todos: Todo[]}`；JSON.parse 在浏览器端执行 |
